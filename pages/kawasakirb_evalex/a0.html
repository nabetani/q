# "4*5+6" => "4*11"
OPS=%w( | & + * )

def solve(x)
  ops = x.scan( /\D+/ ).sort_by{ |op| OPS.index(op) }
  ops.inject(x){ |x,op|
    x.gsub( /\d+#{"\\"+op}\d+/ ){ |part|
        eval(part).to_s
    }
  }
end
